DeleteAllCheckboxes

Phase 2 — strips all 257 form-control checkboxes from MM. Used during initial cleanup.

Phase 2 — done 
Sub DeleteAllCheckboxes()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Meeting Minutes")
    Dim shp As Shape
    Dim count As Long
    count = 0
    Application.ScreenUpdating = False
    For Each shp In ws.Shapes
        If shp.Type = msoFormControl Then
            If shp.FormControlType = xlCheckBox Then
                shp.Delete
                count = count + 1
            End If
        End If
    Next shp
    Application.ScreenUpdating = True
    MsgBox count & " checkboxes deleted from Meeting Minutes."
End Sub

DeleteBrokenCF

Phase 2 — deletes any CF rule whose formula contains #REF!.

Phase 2 — done 
Sub DeleteBrokenCF()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Meeting Minutes")
    Dim fc As FormatCondition
    Dim deleted As Long
    Dim i As Long
    Application.ScreenUpdating = False
    For i = ws.Cells.FormatConditions.Count To 1 Step -1
        Set fc = ws.Cells.FormatConditions(i)
        If InStr(fc.Formula1, "#REF!") > 0 Then
            fc.Delete
            deleted = deleted + 1
        End If
    Next i
    Application.ScreenUpdating = True
    MsgBox deleted & " broken CF rules deleted."
End Sub

Sub DefragmentCF()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Meeting Minutes")
    Application.ScreenUpdating = False
    Dim fc As FormatCondition
    Dim i As Long
    Dim deleted As Long
    For i = ws.Cells.FormatConditions.Count To 1 Step -1
        Set fc = ws.Cells.FormatConditions(i)
        Dim sq As String
        sq = fc.AppliesTo.Address
        If InStr(sq, ",") > 0 Then
            If InStr(sq, "$H$") > 0 Or InStr(sq, "$I$") > 0 Then
                fc.Delete
                deleted = deleted + 1
            End If
        End If
    Next i
    Dim rngP As Range
    Set rngP = ws.Range("I16:I1000")
    AddCellIs rngP, "Critical", RGB(192, 0, 0), RGB(255, 255, 255)
    AddCellIs rngP, "High", RGB(237, 125, 49), RGB(255, 255, 255)
    AddCellIs rngP, "Medium", RGB(255, 192, 0), RGB(26, 26, 46)
    AddCellIs rngP, "Low", RGB(166, 166, 166), RGB(255, 255, 255)
    Dim rngS As Range
    Set rngS = ws.Range("H16:H1000")
    AddCellIs rngS, "Open", RGB(221, 235, 247), RGB(31, 78, 121)
    AddCellIs rngS, "In Progress", RGB(255, 242, 204), RGB(128, 96, 0)
    AddCellIs rngS, "Done", RGB(226, 239, 218), RGB(55, 86, 35)
    AddCellIs rngS, "On Hold", RGB(251, 229, 214), RGB(132, 60, 12)
    AddCellIs rngS, "Waiting", RGB(242, 232, 251), RGB(91, 42, 135)
    Application.ScreenUpdating = True
    MsgBox "Defrag complete. Deleted " & deleted & " fragmented CF rules."
End Sub

Sub AddCellIs(rng As Range, value As String, fillColor As Long, fontColor As Long)
    With rng.FormatConditions.Add(Type:=xlExpression, _
        Formula1:="=" & rng.Cells(1, 1).Address(False, True) & "=""" & value & """")
        .Interior.Pattern = xlSolid
        .Interior.Color = fillColor
        .Font.Color = fontColor
        .Font.Bold = True
        .StopIfTrue = False
    End With
End Sub

Sub CheckCFAfterDefrag()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Meeting Minutes")
    Dim msg As String, i As Long, fc As FormatCondition
    msg = "CF rules on Meeting Minutes:" & vbCrLf & vbCrLf
    For i = 1 To ws.Cells.FormatConditions.Count
        Set fc = ws.Cells.FormatConditions(i)
        Dim sq As String: sq = fc.AppliesTo.Address
        Dim frags As Long: frags = UBound(Split(sq, ",")) + 1
        Dim flag As String: flag = IIf(frags = 1, "OK", "FRAG")
        msg = msg & i & ". [" & flag & "] " & Left(sq, 40) & vbCrLf
        msg = msg & "   " & Left(fc.Formula1, 60) & vbCrLf
    Next i
    MsgBox msg
End Sub

Sub BuildProjectRegister()
    Dim ws As Worksheet
    Application.DisplayAlerts = False
    On Error Resume Next
    ThisWorkbook.Sheets("Project Register").Delete
    On Error GoTo 0
    Application.DisplayAlerts = True
    Set ws = ThisWorkbook.Sheets.Add(After:=ThisWorkbook.Sheets("Summary"))
    ws.Name = "Project Register"
    ws.Range("A1:G1").Value = Array("Project", "Lifecycle", "PM", "Budget", "PC date", "DLP ends", "Notes")
    With ws.Range("A1:G1")
        .Font.Name = "Arial"
        .Font.Size = 10
        .Font.Bold = True
        .Font.Color = RGB(255, 255, 255)
        .Interior.Pattern = xlSolid
        .Interior.Color = RGB(31, 56, 100)
        .HorizontalAlignment = xlLeft
    End With
    ws.Rows(1).RowHeight = 22
    ws.Columns("A").ColumnWidth = 28
    ws.Columns("B").ColumnWidth = 14
    ws.Columns("C").ColumnWidth = 9
    ws.Columns("D").ColumnWidth = 12
    ws.Columns("E").ColumnWidth = 12
    ws.Columns("F").ColumnWidth = 12
    ws.Columns("G").ColumnWidth = 50
    With ws.Range("B2:B100").Validation
        .Delete
        .Add Type:=xlValidateList, Formula1:="Active,DLP,Future,Closed,Strategic,General"
        .IgnoreBlank = True
        .InCellDropdown = True
    End With
    ws.Range("E2:F100").NumberFormat = "dd/mm/yyyy"
    AddLifecycleCF ws.Range("B2:B100"), "Active", RGB(221, 235, 247), RGB(31, 78, 121)
    AddLifecycleCF ws.Range("B2:B100"), "DLP", RGB(225, 245, 238), RGB(15, 110, 86)
    AddLifecycleCF ws.Range("B2:B100"), "Future", RGB(221, 235, 247), RGB(24, 95, 165)
    AddLifecycleCF ws.Range("B2:B100"), "Closed", RGB(241, 239, 232), RGB(95, 94, 90)
    AddLifecycleCF ws.Range("B2:B100"), "Strategic", RGB(238, 237, 254), RGB(83, 74, 183)
    AddLifecycleCF ws.Range("B2:B100"), "General", RGB(241, 239, 232), RGB(68, 68, 65)
    Dim mm As Worksheet
    Set mm = ThisWorkbook.Sheets("Meeting Minutes")
    Dim projects As Object
    Set projects = CreateObject("Scripting.Dictionary")
    Dim r As Long
    For r = 16 To mm.Cells(mm.Rows.Count, 13).End(xlUp).Row
        Dim p As Variant
        p = mm.Cells(r, 13).Value
        If Not IsEmpty(p) And Not IsError(p) Then
            Dim pStr As String
            pStr = Trim(CStr(p))
            If Len(pStr) > 0 And InStr(pStr, "#REF!") = 0 And InStr(pStr, "=IF") = 0 Then
                If Not projects.exists(pStr) Then
                    projects.Add pStr, True
                End If
            End If
        End If
    Next r
    Dim outRow As Long: outRow = 2
    Dim key As Variant
    For Each key In projects.keys
        ws.Cells(outRow, 1).Value = key
        outRow = outRow + 1
    Next key
    ws.Activate
    ws.Range("A2").Select
    ActiveWindow.FreezePanes = True
    MsgBox projects.Count & " projects harvested."
End Sub

Sub ClearColAData()

    lastRow = ws.UsedRange.Rows.Count
    Application.ScreenUpdating = False
    Dim cell As Range

FixODDaysFormulas

Phase 5 — cleared all #REF! formulas from MM col G (OD Days). Cleared 292 cells.

Phase 5 — done 
Sub FixODDaysFormulas()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Meeting Minutes")
    Dim lastRow As Long
    lastRow = ws.UsedRange.Rows.Count
    Application.ScreenUpdating = False
    Dim cleared As Long
    Dim cell As Range
    For Each cell In ws.Range("G16:G" & lastRow)
        If Not cell.MergeCells Then
            Dim hasRef As Boolean
            hasRef = False
            On Error Resume Next
            If InStr(CStr(cell.Formula), "#REF!") > 0 Then hasRef = True
            On Error GoTo 0
            If Not hasRef Then
                If IsError(cell.Value) Then hasRef = True
            End If
            If hasRef Then
                cell.ClearContents
                cleared = cleared + 1
            End If
        End If
    Next cell
    Application.ScreenUpdating = True
    MsgBox "Cleared " & cleared & " #REF! cells in col G."
End Sub

InsertProjectBanners

Phase 6 — walked col I and inserted a Project banner row each time the project changed.

Phase 6 — done 
Sub InsertProjectBanners()
    Dim mm As Worksheet
    Set mm = ThisWorkbook.Sheets("Meeting Minutes")
    Application.ScreenUpdating = False
    Dim lastRow As Long
    lastRow = mm.Cells(mm.Rows.Count, 9).End(xlUp).Row
    Dim r As Long
    Dim prevProj As String
    prevProj = ""
    Dim inserted As Long
    r = 16
    Do While r <= lastRow
        Dim curProj As String
        curProj = Trim(CStr(mm.Cells(r, 9).Value))
        If curProj <> "" And curProj <> prevProj Then
            mm.Rows(r).Insert Shift:=xlDown
            mm.Range("A" & r & ":J" & r).Merge
            With mm.Cells(r, 1)
                .Value = curProj
                .Font.Bold = True
                .Font.Color = RGB(4, 44, 83)
                .Interior.Color = RGB(181, 212, 244)
                .HorizontalAlignment = xlLeft
                .IndentLevel = 2
            End With
            inserted = inserted + 1
            prevProj = curProj
            r = r + 2
            lastRow = lastRow + 1
        Else
            r = r + 1
        End If
    Loop
    Application.ScreenUpdating = True
    MsgBox inserted & " project banners inserted."
End Sub

InsertLifecycleBanners + LCColor

Sub InsertLifecycleBanners()
    Dim mm As Worksheet, pr As Worksheet
    Set mm = ThisWorkbook.Sheets("Meeting Minutes")
    Set pr = ThisWorkbook.Sheets("Project Register")
    Application.ScreenUpdating = False
    Dim lcMap As Object
    Set lcMap = CreateObject("Scripting.Dictionary")
    Dim lastPR As Long
    lastPR = pr.Cells(pr.Rows.Count, 1).End(xlUp).Row
    Dim pr_r As Long
    For pr_r = 2 To lastPR
        Dim pName As String, pLC As String
        pName = Trim(CStr(pr.Cells(pr_r, 1).Value))
        pLC = Trim(CStr(pr.Cells(pr_r, 2).Value))
        If pName <> "" Then lcMap(pName) = pLC
    Next pr_r
    Dim lastRow As Long
    lastRow = mm.Cells(mm.Rows.Count, 9).End(xlUp).Row
    Dim r As Long
    Dim prevLC As String
    prevLC = ""
    Dim inserted As Long
    r = 16
    Do While r <= lastRow
        If mm.Cells(r, 1).MergeCells Then
            r = r + 1
        Else
            Dim curProj As String
            curProj = Trim(CStr(mm.Cells(r, 9).Value))
            If curProj = "" Then
                r = r + 1
            Else
                Dim curLC As String
                If lcMap.exists(curProj) Then
                    curLC = CStr(lcMap(curProj))
                Else
                    curLC = "MISC"
                End If
                If curLC = "" Then curLC = "MISC"
                If curLC <> prevLC Then
                    Dim insertAt As Long
                    insertAt = r - 1
                    If mm.Cells(insertAt, 1).MergeCells Then
                        mm.Rows(insertAt).Insert Shift:=xlDown
                        mm.Range("A" & insertAt & ":J" & insertAt).Merge
                        With mm.Cells(insertAt, 1)
                            .Value = UCase(curLC)
                            .Font.Bold = True
                            .Font.Color = RGB(255, 255, 255)
                            .Interior.Color = LCColor(curLC)
                            .HorizontalAlignment = xlLeft
                            .IndentLevel = 1
                        End With
                        inserted = inserted + 1
                        r = r + 1
                        lastRow = lastRow + 1
                    End If
                    prevLC = curLC
                End If
                r = r + 1
            End If
        End If
    Loop
    Application.ScreenUpdating = True
    MsgBox inserted & " lifecycle banners inserted."
End Sub

Function LCColor(lc As String) As Long
    Select Case lc
        Case "Team":      LCColor = RGB(68, 68, 65)
        Case "Active":    LCColor = RGB(31, 56, 100)
        Case "DLP":       LCColor = RGB(15, 110, 86)
        Case "Future":    LCColor = RGB(24, 95, 165)
        Case "Strategic": LCColor = RGB(83, 74, 183)
        Case "General":   LCColor = RGB(95, 94, 90)
        Case "Closed":    LCColor = RGB(95, 94, 90)
        Case Else:        LCColor = RGB(180, 178, 169)
    End Select
End Function

Sub CleanDataAreaFormatting()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Meeting Minutes")
    Application.ScreenUpdating = False
    Dim lastRow As Long
    lastRow = ws.UsedRange.Rows.Count
    If lastRow < 16 Then lastRow = 16
    Dim cleared As Long
    Dim r As Long
    For r = 16 To lastRow
        If Not ws.Cells(r, 1).MergeCells Then
            ws.Range("A" & r & ":J" & r).Interior.Pattern = xlNone
            cleared = cleared + 1
        End If
    Next r
    Dim dataArea As Range
    Set dataArea = ws.Range("A15:J" & lastRow)
    With dataArea.Borders
        .LineStyle = xlContinuous
        .Weight = xlThin
        .Color = RGB(212, 212, 212)
    End With
    Application.ScreenUpdating = True
    MsgBox "Cleared fills on " & cleared & " data rows."
End Sub
